def tri_selection(t):
    N = len(t)
    i = 0
    while i < N-1 :
        j = i + 1
        min = i
        while j < N:
            if t[j]<t[min]:
                min=j
            j = j + 1
        if min != i :
            k = t[i]
            t[i] = t[min]
            t[min] = k
        i = i +1
    return t
